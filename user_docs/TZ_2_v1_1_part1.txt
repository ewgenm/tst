### 1.2. Принципы кода
*   **Type-Safe:** Все API responses, props, emits, store state — строго типизированы по ТЗ №0.
*   **Composition API:** Только `<script setup lang="ts">`, никаких Options API.
*   **Headless UI:** Radix Vue для доступных компонентов + кастомные стили Tailwind.
*   **Reactive State:** Pinia для глобального состояния, composables для локальной логики.
*   **Optimistic Updates с откатом:** UI обновляется сразу, при ошибке — откат изменений + toast.
*   **Error Boundaries:** Глобальная обработка ошибок API с toast-уведомлениями и field-level валидацией.
*   **Test-Driven:** Vitest + Vue Test Utils для критических компонентов и stores.

```typescript
import axios from 'axios'
import type { AxiosError, InternalAxiosRequestConfig, AxiosResponse } from 'axios'
import { useAuthStore } from '@/stores/auth'
import { useToast } from '@/composables/useToast'

export const apiClient = axios.create({
  baseURL: import.meta.env.VITE_API_URL || '/api/v1',
  headers: {
    'Content-Type': 'application/json',
    'Accept': 'application/json',
    'X-Requested-With': 'XMLHttpRequest',
  },
  withCredentials: true, // Для Sanctum cookies (CRITICAL FIX)
})

// Request interceptor - CSRF token
apiClient.interceptors.request.use(async (config: InternalAxiosRequestConfig) => {
  if (!config.headers.get('X-XSRF-TOKEN')) {
    // Получить CSRF токен
    await axios.get('/sanctum/csrf-cookie', { withCredentials: true })
  }
  return config
})

// Response interceptor - error handling with retry logic (CRITICAL FIX)
apiClient.interceptors.response.use(
  (response) => response,
  async (error: AxiosError) => {
    const { show } = useToast()
    const originalRequest = error.config as InternalAxiosRequestConfig & { _retry?: boolean }

    // Rate limiting - retry after delay (CRITICAL FIX #17)
    if (error.response?.status === 429 && !originalRequest._retry) {
      const retryAfter = error.response.headers['retry-after'] || '60'
      originalRequest._retry = true
      
      show(`Слишком много запросов. Повтор через ${retryAfter} сек.`, 'warning')
      await new Promise(resolve => setTimeout(resolve, Number(retryAfter) * 1000))
      return apiClient(originalRequest)
    }

    if (error.response?.status === 401) {
      const authStore = useAuthStore()
      authStore.logout()
      window.location.href = '/login'
      show('Сессия истекла. Пожалуйста, войдите снова.', 'error')
    }

    if (error.response?.status === 403) {
      show('У вас нет доступа к этому ресурсу.', 'error')
    }

    // Field-level validation errors (CRITICAL FIX #6)
    if (error.response?.status === 422) {
      const data = error.response.data as { error?: { message: string; details?: Record<string, string[]> } }
      
      if (data?.error?.details) {
        // Пробрасываем детали валидации для vee-validate
        window.dispatchEvent(new CustomEvent('validation-errors', { 
          detail: data.error.details 
        }))
      }
      
      show(data?.error?.message || 'Ошибка валидации.', 'error')
    }

    if (error.response?.status === 429 && !originalRequest._retry) {
      show('Слишком много запросов. Попробуйте позже.', 'error')
    }

    return Promise.reject(error)
  }
)
```

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { User } from '@/types'
import { apiClient } from '@/api/client'

export const useAuthStore = defineStore('auth', () => {
  const user = ref<User | null>(null)
  const isLoading = ref(false)

  const isAuthenticated = computed(() => !!user.value)
  const isEmailVerified = computed(() => !!user.value?.email_verified_at)
  const userTimezone = computed(() => user.value?.timezone || 'UTC')
  const userLocale = computed(() => user.value?.locale || 'ru')

  async function fetchMe() {
    try {
      const { data } = await apiClient.get('/auth/me')
      user.value = data.data
      // Apply theme from user profile
      const uiStore = useUIStore()
      if (user.value?.theme) {
        uiStore.setTheme(user.value.theme)
      }
    } catch {
      user.value = null
    }
  }

  async function login(email: string, password: string) {
    isLoading.value = true
    try {
      const { data } = await apiClient.post('/auth/login', { email, password })
      user.value = data.data.user
      return { success: true }
    } catch (error: any) {
      return { 
        success: false, 
        error: error.response?.data?.error?.message || 'Ошибка входа' 
      }
    } finally {
      isLoading.value = false
    }
  }

  async function register(name: string, email: string, password: string) {
    isLoading.value = true
    try {
      const { data } = await apiClient.post('/auth/register', { name, email, password })
      user.value = data.data.user
      return { success: true }
    } catch (error: any) {
      return { 
        success: false, 
        error: error.response?.data?.error?.message || 'Ошибка регистрации' 
      }
    } finally {
      isLoading.value = false
    }
  }

  function logout() {
    user.value = null
    apiClient.post('/auth/logout')
  }

  async function updateProfile(payload: Partial<User>) {
    const { data } = await apiClient.put('/auth/me', payload)
    user.value = data.data
  }

  return {
    user,
    isLoading,
    isAuthenticated,
    isEmailVerified,
    userTimezone,
    userLocale,
    fetchMe,
    login,
    register,
    logout,
    updateProfile,
  }
})
```

### 3.2. Tasks Store (с откатами и project channels)
**`src/stores/tasks.ts`:**
```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import type { Task, TaskFilters } from '@/types'
import { apiClient } from '@/api/client'
import { useWebSocket } from '@/composables/useWebSocket'
import { useToast } from '@/composables/useToast'

export const useTasksStore = defineStore('tasks', () => {
  const tasks = ref<Task[]>([])
  const isLoading = ref(false)
  const pagination = ref({ current_page: 1, per_page: 20, total: 0, total_pages: 0, has_more: false })
  const filters = ref<TaskFilters>({})
  const { show } = useToast()

  const inboxTasks = computed(() => tasks.value.filter(t => !t.project_id && t.status !== 'done'))
  const todayTasks = computed(() => {
    const today = new Date().toISOString().split('T')[0]
    return tasks.value.filter(t => t.due_at?.startsWith(today))
  })

  async function fetchTasks(params: TaskFilters = {}) {
    isLoading.value = true
    try {
      const { data } = await apiClient.get('/tasks', { params })
      tasks.value = data.data
      pagination.value = data.pagination
      filters.value = params
    } finally {
      isLoading.value = false
    }
  }

  // Optimistic update с откатом (CRITICAL FIX #7)
  async function createTask(payload: Partial<Task>) {
    const tempId = Date.now()
    const tempTask: Task = {
      ...payload,
      id: tempId,
      created_at: new Date().toISOString(),
      updated_at: new Date().toISOString(),
      deleted_at: null,
    } as Task

    // Optimistic add
    tasks.value.unshift(tempTask)

    try {
      const { data } = await apiClient.post('/tasks', payload)
      // Replace temp with real task
      const index = tasks.value.findIndex(t => t.id === tempId)
      if (index !== -1) tasks.value[index] = data.data
      return data.data
    } catch (error: any) {
      // Rollback
      tasks.value = tasks.value.filter(t => t.id !== tempId)
      show('Не удалось создать задачу', 'error')
      throw error
    }
  }

  async function updateTask(id: number, payload: Partial<Task>) {
    const original = tasks.value.find(t => t.id === id)
    const originalData = { ...original }

    // Optimistic update
    const index = tasks.value.findIndex(t => t.id === id)
    if (index !== -1) {
      tasks.value[index] = { ...tasks.value[index], ...payload }
    }

    try {
      const { data } = await apiClient.put(`/tasks/${id}`, payload)
      tasks.value[index] = data.data
      return data.data
    } catch (error: any) {
      // Rollback
      if (index !== -1 && originalData) {
        tasks.value[index] = originalData
      }
      show('Не удалось обновить задачу', 'error')
      throw error
    }
  }

  async function deleteTask(id: number) {
    const originalIndex = tasks.value.findIndex(t => t.id === id)
    const originalTask = tasks.value[originalIndex]

    // Optimistic remove
    tasks.value = tasks.value.filter(t => t.id !== id)

    try {
      await apiClient.delete(`/tasks/${id}`) // Returns 204
    } catch (error: any) {
      // Rollback
      if (originalIndex !== -1 && originalTask) {
        tasks.value.splice(originalIndex, 0, originalTask)
      }
      show('Не удалось удалить задачу', 'error')
      throw error
    }
  }

  // Handle recurring task creation (CRITICAL FIX #12)
  async function completeTask(id: number) {
    const originalIndex = tasks.value.findIndex(t => t.id === id)
    const originalTask = tasks.value[originalIndex]

    try {
      const { data } = await apiClient.post(`/tasks/${id}/complete`)
      
      // Update the completed task
      if (originalIndex !== -1) {
        tasks.value[originalIndex] = data.data
      }

      // Check if new recurring task was created (different ID or new task in response)
      // Backend may return the completed task OR include next occurrence
      // For MVP, we'll refetch to ensure consistency
      await fetchTasks(filters.value)
      
      show('Задача выполнена!', 'success')
      return data.data
    } catch (error: any) {
      show('Не удалось завершить задачу', 'error')
      throw error
    }
  }

  async function reorderTask(id: number, position: number) {
    const original = tasks.value.find(t => t.id === id)
    const originalPosition = original?.position

    // Optimistic update
    if (original) original.position = position

    try {
      await apiClient.put(`/tasks/${id}/position`, { position }) // CRITICAL FIX #8
    } catch (error: any) {
      // Rollback
      if (original) original.position = originalPosition || 0
      show('Не удалось изменить порядок', 'error')
      throw error
    }
  }

  // WebSocket subscription for projects (CRITICAL FIX #5)
  function setupRealtime(userId?: number, projectId?: number) {
    const { subscribe } = useWebSocket()
    
    // Personal notifications
    if (userId) {
      subscribe(`private-user.${userId}`, 'task.created', (payload: { task: Task }) => {
        tasks.value.unshift(payload.task)
      })
      subscribe(`private-user.${userId}`, 'task.updated', (payload: { task: Task }) => {
        const index = tasks.value.findIndex(t => t.id === payload.task.id)
        if (index !== -1) tasks.value[index] = payload.task
      })
      subscribe(`private-user.${userId}`, 'task.deleted', (payload: { task_id: number }) => {
        tasks.value = tasks.value.filter(t => t.id !== payload.task_id)
      })
    }

    // Project-specific updates (CRITICAL FIX #5)
    if (projectId) {
      subscribe(`private-project.${projectId}`, 'task.created', (payload: { task: Task }) => {
        if (!tasks.value.find(t => t.id === payload.task.id)) {
          tasks.value.unshift(payload.task)
        }
      })
      subscribe(`private-project.${projectId}`, 'task.updated', (payload: { task: Task }) => {
        const index = tasks.value.findIndex(t => t.id === payload.task.id)
        if (index !== -1) tasks.value[index] = payload.task
      })
      subscribe(`private-project.${projectId}`, 'comment.created', (payload: { comment: any }) => {
        // Handle comment updates
      })
    }
  }

  return {
    tasks,
    isLoading,
    pagination,
    filters,
    inboxTasks,
    todayTasks,
    fetchTasks,
    createTask,
    updateTask,
    deleteTask,
    completeTask,
    reorderTask,
    setupRealtime,
  }
})
```
