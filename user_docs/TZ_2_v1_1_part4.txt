import { describe, it, expect, vi } from 'vitest'
import { mount } from '@vue/test-utils'
import TaskItem from '../TaskItem.vue'
import type { Task } from '@/types'

const mockTask: Task = {
  id: 1,

  deleted_at: null,
}

describe('TaskItem', () => {
  it('renders task title correctly', () => {
    const wrapper = mount(TaskItem, { props: { task: mockTask } })
    expect(wrapper.text()).toContain('Test Task')
  })

  it('emits toggle event on checkbox click', async () => {

## 10. ДЕПЛОЙ И ОПТИМИЗАЦИЯ

### 10.1. Production build
```bash
npm run build
# Output: dist/ с оптимизированными бандлами
```

### 10.2. Оптимизации
*   **Code splitting:** Lazy loading роутов (`component: () => import()`).
*   **Tree shaking:** Unused exports удаляются автоматически.
*   **Image optimization:** Использовать WebP, lazy loading.
*   **PWA:** Service Worker для offline доступа (через Vite PWA plugin).
*   **Virtual scroll:** `vue-virtual-scroller` для списков >100 элементов.
*   **Skeleton loaders:** Для всех async состояний.

### 10.3. Capacitor (Mobile/Desktop)
```bash
npm install @capacitor/core @capacitor/cli @capacitor/android @capacitor/ios
npx cap init TaskSyncTeam com.tasksync.app
npx cap add android
npx cap add ios
npx cap sync
```

**Для Windows (Tauri):**
```bash
npm install @tauri-apps/cli
npx tauri init
```
