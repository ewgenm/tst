```php
class TaskCreated implements ShouldBroadcastNow

    {
```

```php
Broadcast::channel('project.{projectId}', function ($user, $projectId) {
    $project = \App\Models\Project::find($projectId);
    if (!$project) return false;
    
    // Владелец ИЛИ участник с active статусом
    return $project->owner_id === $user->id 
        || $project->members()
            ->where('user_id', $user->id)
            ->where('status', 'active')
            ->exists();
});

Broadcast::channel('user.{userId}', function ($user, $userId) {
    return $user->id === (int) $userId;
});
```

```php
public function generateSubtasks(GenerateSubtasksRequest $request)
{
    if (RateLimiter::tooManyAttempts('ai:'.$request->user()->id, 20)) {
        return response()->json([
            'success' => false,
            'error' => ['code' => 'RATE_LIMIT_EXCEEDED', 'message' => 'Превышен лимит AI запросов (20/час)']
        ], 429);
    }
    
    RateLimiter::hit('ai:'.$request->user()->id, 3600);
    
    $subtasks = $this->aiService->generateSubtasks($request->input('description'));
    
    return response()->json(['success' => true, 'data' => ['subtasks' => $subtasks]]);
}
```

```php
public function test_user_can_create_task_in_project(): void
{
    $user = User::factory()->create(['email_verified_at' => now()]);
    $project = Project::factory()->for($user, 'owner')->create();
    
    $response = $this->actingAs($user)
        ->postJson('/api/v1/tasks', [
            'title' => 'New Task',
            'project_id' => $project->id,
            'status' => 'todo',
            'priority' => 'medium',
        ]);
    
    $response->assertStatus(201)
        ->assertJsonStructure(['success', 'data' => ['id', 'title', 'project_id']]);
    
    $this->assertDatabaseHas('tasks', ['title' => 'New Task', 'project_id' => $project->id]);
}

public function test_subtasks_are_cascade_deleted(): void
{
    $user = User::factory()->create(['email_verified_at' => now()]);
    $parent = Task::factory()->for($user, 'creator')->create();
    $subtask = Task::factory()->for($user, 'creator')->create(['parent_task_id' => $parent->id]);
    
    $parent->delete();
    
    $this->assertDatabaseMissing('tasks', ['id' => $parent->id]);
    $this->assertDatabaseMissing('tasks', ['id' => $subtask->id]); // Каскадное удаление
}

public function test_delete_returns_204(): void
{
    $user = User::factory()->create(['email_verified_at' => now()]);
    $task = Task::factory()->for($user, 'creator')->create();
    
    $response = $this->actingAs($user)
        ->deleteJson("/api/v1/tasks/{$task->id}");
    
    $response->assertStatus(204);
}
```
